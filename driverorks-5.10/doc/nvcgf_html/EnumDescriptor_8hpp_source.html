136template <typename EnumT>
137auto mapEnumNameToValue(dw::core::StringView const& name) -> EnumT
138{
139 const auto pairs = EnumDescription<EnumT>::get();
140 for (const auto& pair : pairs)
141 {
142 if (pair.first == name)
143 {
144 return pair.second;
145 }
146 }
147 FixedString<4096> validNames{};
148 for (const auto& pair : pairs)
149 {
150 validNames << " " << pair.first;
151 }
152 throw ExceptionWithStatus(DW_INVALID_ARGUMENT, "Invalid enumerator name '", name, "' for enum type ", typeid(EnumT).name(), ", valid names:", validNames);
153}

162template <typename EnumT>
163dw::core::StringView mapEnumValueToName(EnumT value)
164{
165 const auto pairs = EnumDescription<EnumT>::get();
166 for (const auto& pair : pairs)
167 {
168 if (pair.second == value)
169 {
170 return pair.first;
171 }
172 }
173 throw ExceptionWithStatus(DW_INVALID_ARGUMENT, "The enumerator value is invalid: ", typeid(EnumT).name(), " ", static_cast<std::underlying_type_t<EnumT>>(value));
174}
