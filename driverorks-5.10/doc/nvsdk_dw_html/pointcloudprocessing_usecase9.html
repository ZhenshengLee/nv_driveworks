Point Cloud Motion Compensator

This tutorial demonstrates how to compensate motion in input point cloud.

Create motion compensator module:

dwMotionCompensatorParams params{};
dwMotionCompensator_getDefaultParams(&params, &lidarProperties);
dwMotionCompensatorHandle_t compensator{};
dwMotionCompensator_initialize(&compensator, &params, context);

Create output point cloud and bind it to compensator.

// Let `inputPointCloud` - accumulated input point cloud
// `timestampChannelIndex` - index of auxiliary channel in input point cloud, containing
// individual point timestamps
// `motionState` - egomotion state
// Bind input to compensator
dwMotionCompensator_bindInput(&inputPointCloud, timestampChannelIndex, compensator);
// Bind motion state
dwMotionCompensator_bindEgomotionState(motionState, compensator);
// Create output point cloud
dwPointCloud outputPointCloud{};
dwPointCloud_create(&outputPointCloud, // output point cloud
DW_POINTCLOUD_FORMAT_XYZI, // format of a point, xyzi of rthi
DW_MEMORY_TYPE_CUDA, // memory type, CUDA or CPU
inputPointCloud.maxPointsPerLayer, // maximum number of points in single lidar return
inputpointCloud.numLayers, // number of requested returns
nullptr, // size of aux channel elements
0U); // number of auxiliary channels
// Bind output point cloud
dwMotionCompensator_bindOutput(&outputPointCloud, compensator);
// Set reference timestamp for motion model. By default host end timestamp of input point cloud is used
dwMotionCompensator_setReferenceTimestamp(timestamp, compensator);
// Set CUDA stream
dwMotionCompensator_setCUDAStream(stream, compensator);
